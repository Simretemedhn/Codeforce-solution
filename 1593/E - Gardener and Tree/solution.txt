from collections import defaultdict, deque 
 
t = int(input())
 
for _ in range(t):
    graph = defaultdict(list)
    in_or_out_degree = {}
    blank = input()
    vertex, oper = map(int, input().split())
    
    for _ in range(vertex-1):
        src, dst = map(int, input().split())
        in_or_out_degree[src] = in_or_out_degree.get(src, 0) + 1
        in_or_out_degree[dst] = in_or_out_degree.get(dst, 0) + 1
        graph[src].append(dst)
        graph[dst].append(src)
    
    if oper >= vertex:
        print(0)
        continue
    
    q = deque([node for node, val in in_or_out_degree.items() if val == 1])
    removed_round = {}
    for node in q:
        removed_round[node] = 1
    
    level = 0
    while q and level < oper:
        level += 1
        n = len(q)
        
        for _ in range(n):
            node = q.popleft()
            
            for nei in graph[node]:
                if nei not in removed_round:
                    in_or_out_degree[nei] -= 1
                    if in_or_out_degree[nei] == 1:
                        q.append(nei)
                        removed_round[nei] = level + 1
    
    remaining = 0
    for i in range(1, vertex + 1):
        if i not in removed_round or removed_round[i] > oper:
            remaining += 1
    
    print(remaining)
