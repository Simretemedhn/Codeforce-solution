from collections import Counter t = int(input()) for _ in range(t):    n, l, r = map(int, input().split())    socks = list(map(int, input().split()))     left = socks[:l]    right = socks[l:]     left_count = Counter(left)    right_count = Counter(right)     for color in list(left_count.keys()):        if color in right_count:            pairs = min(left_count[color], right_count[color])            left_count[color] -= pairs            right_count[color] -= pairs            if left_count[color] == 0:                del left_count[color]            if right_count[color] == 0:                del right_count[color]     left_unmatched = sum(left_count.values())    right_unmatched = sum(right_count.values())     cost = 0     if left_unmatched > right_unmatched:        diff = (left_unmatched - right_unmatched) // 2        converted = 0        for color in list(left_count.keys()):            while left_count[color] >= 2 and converted < diff:                left_count[color] -= 2                converted += 1                cost += 1            if left_count[color] == 0:                del left_count[color]        left_unmatched = sum(left_count.values())        cost += left_unmatched     elif right_unmatched > left_unmatched:        diff = (right_unmatched - left_unmatched) // 2        converted = 0        for color in list(right_count.keys()):            while right_count[color] >= 2 and converted < diff:                right_count[color] -= 2                converted += 1                cost += 1            if right_count[color] == 0:                del right_count[color]        right_unmatched = sum(right_count.values())        cost += right_unmatched     else:        cost += left_unmatched       print(cost)
